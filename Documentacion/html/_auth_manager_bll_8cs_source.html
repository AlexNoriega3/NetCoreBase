110 private async Task<string> GenerateToken()
111 {
112 var securitykey = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(_configuration["JwtSettings:Key"]));
113
114 var credentials = new SigningCredentials(securitykey, SecurityAlgorithms.HmacSha256);
115
116 var roles = await _userManager.GetRolesAsync(_user);
117 var roleClaims = roles.Select(x => new Claim(ClaimTypes.Role, x)).ToList();
118 var userClaims = await _userManager.GetClaimsAsync(_user);
119
120 var claims = new List<Claim>
121 {
122 new Claim(JwtRegisteredClaimNames.Sub, _user.Email),
123 new Claim(JwtRegisteredClaimNames.Jti, Guid.NewGuid().ToString()),
124 new Claim(JwtRegisteredClaimNames.Email, _user.Email),
125 new Claim("uid", _user.Id),
126 }
127 .Union(userClaims).Union(roleClaims);
128
129 var token = new JwtSecurityToken(
130 issuer: _configuration["JwtSettings:Issuer"],
131 audience: _configuration["JwtSettings:Audience"],
132 claims: claims,
133 expires: DateTime.Now.AddMinutes(Convert.ToInt32(_configuration["JwtSettings:DurationInMinutes"])),
134 signingCredentials: credentials
135 );
136
137 return new JwtSecurityTokenHandler().WriteToken(token);
138 }
